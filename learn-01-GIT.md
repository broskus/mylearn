##<GIT - ШПАРГАЛКА>##

## 🔧 Настройка:

git config --global user.name "Имя"           # Кто ты (для коммитов)
git config --global user.email "email@mail"   # Твоя почта
git config --global core.editor "nano"        # Редактор для сообщений
git config --global init.defaultBranch main   # Имя главной ветки
git config --global alias.lg "log --oneline --graph --all --decorate"  # Красивый лог
git config --global --list                    # Посмотреть все настройки

##BAZA
git init                    # Создать репозиторий в текущей папке
git status                  # Что происходит? (ветка, изменённые файлы)
git add <файл>              # Подготовить файл к коммиту (в staging)
git add .                   # Подготовить все файлы сразу
git commit -m "сообщение"   # Сохранить снимок изменений в истории
git log                     # Посмотреть историю коммитов
git lg                      # Красивая история (наш алиас)
git diff                    # Что изменилось в файлах (до git add)
git diff --staged           # Что подготовлено к коммиту (после git add)

#🌿ВЕТКИ
git branch                  # Список всех локальных веток
git branch <имя>            # Создать новую ветку
git checkout <ветка>        # Переключиться на ветку
git checkout -b <ветка>     # Создать ветку И переключиться на неё
git switch <ветка>          # Переключиться (новая, более понятная команда)
git switch -c <ветка>       # Создать и переключиться (новая команда)
git merge <ветка>           # Влить изменения из указанной ветки в текущую
git rebase <ветка>          # Перенести свои коммиты поверх указанной ветки
git branch -d <ветка>       # Удалить ветку (если она уже слита)
git branch -D <ветка>       # Принудительно удалить ветку (даже если не слита)

#🌐Удалённый репозиторий (GitHub)
git remote -v               # Показать адреса удалённых репозиториев
git remote add origin <url> # Привязать удалённый репозиторий (обычно называют origin)
git remote remove origin    # Отвязать удалённый репозиторий
git clone <url>             # Скопировать удалённый репозиторий к себе
git push -u origin main     # Первый пуш: отправить + запомнить связь с origin/main
git push                    # Отправить свои коммиты на GitHub
git fetch                   # Скачать изменения с GitHub, но НЕ сливать их
git pull                    # Скачать И сразу слить изменения с GitHub
git push origin --delete <ветка>  # Удалить ветку на GitHub

#↩️Отмена изменений
git restore <файл>              # Отменить изменения в файле (вернуть к последнему коммиту)
git restore --staged <файл>     # Убрать файл из staging (отменить git add)
git reset --soft HEAD~1         # Отменить последний коммит, изменения оставить в staging
git reset HEAD~1                # Отменить коммит, изменения оставить в файлах (но не в staging)
git reset --hard HEAD~1         # Отменить коммит И удалить все изменения (ОПАСНО!)
git revert HEAD                 # Создать новый коммит, который отменяет последний (безопасно)
git revert <хеш>                # Отменить конкретный коммит по его хешу

#🚀Продвинутое
git stash                   # Спрятать недоделанные изменения (чтобы переключить ветку)
git stash list              # Посмотреть список спрятанных изменений
git stash pop               # Достать последние спрятанные изменения и применить
git stash apply stash@{N}   # Достать конкретный stash по номеру
git stash drop              # Удалить последний stash без применения
git stash clear             # Удалить все stash

git cherry-pick <хеш>       # Взять один конкретный коммит из другой ветки и применить здесь
git reflog                  # История всех перемещений HEAD (спасает потерянные коммиты!)
git rebase -i HEAD~5        # Интерактивный rebase: переписать последние 5 коммитов

#🔥Конфликты при слиянии
# Если Git сказал "CONFLICT":
nano <файл>                 # 1. Открыть файл, найти маркеры <<<<<<<, =======, >>>>>>>
                            # 2. Выбрать нужную версию, удалить маркеры, сохранить
git add <файл>              # 3. Сказать Git, что конфликт решён
git merge --continue        # 4. Продолжить merge
git cherry-pick --continue  #    или продолжить cherry-pick
git rebase --continue       #    или продолжить rebase

# Если передумал:
git merge --abort           # Отменить merge
git cherry-pick --abort     # Отменить cherry-pick
git rebase --abort          # Отменить rebase

#📌Полезные мелочи
git log --oneline           # Коммиты в одну строку
git log -n 5                # Только последние 5 коммитов
git log --graph             # История с графиком веток
git log -p                  # История с показом изменений в каждом коммите
git show <хеш>              # Показать детали конкретного коммита
git blame <файл>            # Кто и когда менял каждую строку файла
git shortlog -sn            # Сколько коммитов сделал каждый автор

#🎯Типичный рабочий день
git pull                    # Получить свежие изменения
git checkout -b feature-x   # Создать ветку для новой фичи
# ... работа ...
git add .                   # Подготовить всё
git commit -m "feat: ..."   # Закоммитить
git push -u origin feature-x  # Отправить ветку на GitHub
# ... создать PR на GitHub, пройти ревью, слить ...
git checkout main           # Вернуться в main
git pull                    # Получить свежий main со слитым PR
git branch -d feature-x     # Удалить локальную ветку (она больше не нужна)

#💡 Золотые правила
Коммить часто — маленькие коммиты легко откатить
Пиши понятные сообщения — через месяц сам себе скажешь спасибо
Не коммить в main напрямую — всегда через ветки
Проверяй git status — перед каждым коммитом
Делай git pull перед работой — чтобы не было конфликтов
Не используй git push --force в общих ветках — сломаешь историю коллегам
Не коммить пароли и ключи — используй .gitignore
